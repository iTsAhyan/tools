import os
import zipfile
import subprocess
from google.colab import files

def run_uploader():
    print("=========================================")
    print("  AHYAN CREATIONS LTD - ZIP To GITHUB    ")
    print("=========================================")
    
    print("\n[+] STEP 1: Please select the project ZIP file from your device:")
    try:
        uploaded = files.upload()
    except Exception as e:
        print(f"\n[!] Browser Environment Error: {e}")
        return

    if not uploaded:
        print("[!] Error: No file was uploaded. Process terminated!")
        return

    zip_name = list(uploaded.keys())[0]
    extract_dir = '/content/extracted_project'

    if os.path.exists(extract_dir):
        subprocess.run(f"rm -rf {extract_dir}", shell=True)

    with zipfile.ZipFile(zip_name, 'r') as zip_ref:
        zip_ref.extractall(extract_dir)

    print("[+] File extraction successful!")

    print("\n-----------------------------------------")
    print("  STEP 2: Enter Your GitHub Information  ")
    print("-----------------------------------------")
    
    token = input("[?] Enter your GitHub Personal Access Token: ").strip()
    repo = input("[?] Enter your target GitHub Repository Name: ").strip()
    user = input("[?] Enter your GitHub Username: ").strip()
    email = input("[?] Enter your GitHub Email: ").strip()

    if not token or not repo or not user or not email:
        print("\n[!] Error: All GitHub fields (Token, Repo, Username, Email) are strictly required!")
        return

    target_dir = extract_dir
    subdirs = [os.path.join(extract_dir, d) for d in os.listdir(extract_dir) if os.path.isdir(os.path.join(extract_dir, d))]
    if subdirs:
        target_dir = subdirs[0]

    def run_cmd(cmd, cwd=None):
        return subprocess.run(cmd, shell=True, check=True, cwd=cwd, stdout=subprocess.DEVNULL, stderr=subprocess.PIPE)

    print("\n[+] Pushing your project files to GitHub (Please wait)...")
    try:
        subprocess.run("git init", shell=True, cwd=target_dir, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
        subprocess.run(f"git config --global user.name '{user}'", shell=True, cwd=target_dir, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
        subprocess.run(f"git config --global user.email '{email}'", shell=True, cwd=target_dir, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
        
        subprocess.run("git checkout -b main", shell=True, cwd=target_dir, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
        subprocess.run("git add .", shell=True, cwd=target_dir, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
        subprocess.run("git commit -m 'Project uploaded via Ahyan Creations Engine'", shell=True, cwd=target_dir, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
        
        remote_url = f"https://{token}@github.com/{user}/{repo}.git"
        subprocess.run(f"git remote add origin {remote_url}", shell=True, cwd=target_dir, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
        
        # পুশ করার সময় এরর মেসেজ ট্র্যাক করার ব্যবস্থা
        push_res = subprocess.run("git push -u origin main --force", shell=True, cwd=target_dir, stdout=subprocess.DEVNULL, stderr=subprocess.PIPE)
        
        if push_res.returncode == 0:
            print("\n🎉 Congratulations! Your project files have been successfully uploaded to GitHub.")
            print(f"🔗 Repository Link: https://github.com/{user}/{repo}")
            print("=========================================")
        else:
            error_msg = push_res.stderr.decode().strip()
            print("\n[!] Upload failed! GitHub rejected the request.")
            print("👉 Please ensure that:")
            print("   1. Your GitHub Token is valid and has 'repo' permissions.")
            print("   2. You have already created the repository on GitHub with the exact same name.")
            print(f"\n[Technical Details]: {error_msg}")
            
    except Exception as e:
        print(f"\n[!] An unexpected error occurred: {e}")

run_uploader()
